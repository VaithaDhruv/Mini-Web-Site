import random

j = 1
a = random.randint(1, 100)
while True:
    n = int(input("Enter a number between 1 to 100: "))
    if n > a:
        print("Enter a smaller number: ")
        j += 1
    elif n < a:
        print("Enter a greater number: ")
        j += 1
    else:
        print("Congratulations! You Have Guessed The Number.")
        break
print(f"Thanks for playing the game. You guessed the number in {j} attempts.")
