i = int(input("Enter how many numbers you want to enter: "))
k = []

for j in range(i):
    if j == 0:
        k.append(int(input(f"Enter the {j+1}st number: ")))
    elif j == 1:
        k.append(int(input(f"Enter the {j+1}nd number: ")))
    elif j == 2:
        k.append(int(input(f"Enter the {j+1}rd number: ")))
    else:
        k.append(int(input(f"Enter the {j+1}th number: ")))

while True:
    g = input(f"Enter the operation (+,-,/,*,**,//,%) you want to do with all {i} numbers (or type 'exit'): ")

    if g == "exit":
        print("Goodbye!")
        break

    s = k[0]

    if g == "+":
        for e in range(1, i):
            s = s + k[e]
        print("Result =", s)

    elif g == "-":
        for e in range(1, i):
            s = s - k[e]
        print("Result =", s)

    elif g == "*":
        for e in range(1, i):
            s = s * k[e]
        print("Result =", s)

    elif g == "/":
        for e in range(1, i):
            s = s / k[e]
        print("Result =", s)

    elif g == "//":
        for e in range(1, i):
            s = s // k[e]
        print("Result =", s)

    elif g == "%":
        for e in range(1, i):
            s = s % k[e]
        print("Result =", s)

    elif g == "**":
        for e in range(1, i):
            s = s ** k[e]
        print("Result =", s)

    else:
        print("Invalid input, please input again.")
