import random

player_score = 0
computer_score = 0
while True:
    i=input("Enter either rock , paper or scissor:").lower()
    r=random.randint(1,3)
    k=""
    if r==1:
        k="rock"
    elif r==2:
        k="paper"
    else:
        k="scissor"
    print(f"Computer:{k}\n")
        
    if i=="scissor" and k=="scissor":
        print("Draw")
    elif i=="scissor" and k=="rock":
        print("You loose!")
        computer_score = computer_score + 1
    elif i=="scissor" and k=="paper":
        print("Congratulations! You Won.")
        player_score = player_score + 1
    elif i=="rock" and k=="rock":
        print("Draw")
    elif i=="rock" and k=="paper":
        print("You loose!")
        computer_score = computer_score + 1
    elif i=="rock" and k=="scissor":
        print("Congratulations! You Won.")
        player_score = player_score + 1
    elif i=="paper" and k=="paper":
        print("Draw")
    elif i=="paper" and k=="scissor":
        print("You loose!")
        computer_score = computer_score + 1
    elif i=="paper" and k=="rock":
        print("Congratulations! You Won.")
        player_score = player_score + 1
    else:
        print("There seems to be mistake please try again.")
        
    x=input("Do You Want To Play Again? (y/n).\n").lower()
    if x=="y":
        continue
    elif x=="n":
        print("Thank You for playing the game.Goodbye!")
        break
    else:
        print("Please enter valid input.")
